FROM INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_SCHEMA = 'modelinvoke'
  AND TABLE_NAME = 'user_api_key'

### 7.2 如果 claude-sonnet-4.5 仍然报错

```sql
-- 检查模型覆盖规则是否生效
SELECT * FROM model_override_rule
WHERE source_pattern = 'claude-sonnet-4.5' AND enabled = 1;

-- 检查目标模型是否存在
SELECT * FROM unified_model WHERE id = 4;

-- 检查模型-渠道映射
SELECT m.*, c.name, c.is_healthy, c.priority
FROM model_channel_mapping m
JOIN channel c ON m.channel_id = c.id
WHERE m.unified_model_id = 4 AND m.enabled = 1
ORDER BY c.priority;
```
